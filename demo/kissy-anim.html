实例一：Anim (elem, props[, duration, easing, completeFn])

		KISSY.use('anim, event', function(S, Anim, Event) {
			Event.on('#button1', 'click', function(event) {
				var anim = new Anim('.demo1', "width:200px; height:200px", 1, S.Easing.easeNone);
				anim.run();
			});
		});

实例四：使用回调函数，当第一个动画执行完毕后，执行回调中的第二个动画

		KISSY.use('anim, event', function(S, Anim, Event) {
			Event.on('#button4', 'click', function(event) {
				var anim = new Anim('.demo4', {'width': '200px', 'height': '200px'}, 1, S.Easing.bounceOut, function() {
					var anim2 = new Anim('.demo4', {'width': '300px', 'height': '50px', 'background': 'red'}, S.Easing.bounceOut);
					anim2.run();
				});
				anim.run();
			});
		});

实例六：使用Anim的成员方法

		KISSY.use('anim, event', function(S, Anim, Event) {
			Event.on('#button6', 'click', function(event) {
				var anim7 = new Anim('.demo6', "width:200px; height:200px", 10, S.Easing.easeNone);
				anim7.run();
				Event.on('#button7', 'click', function(event) {
					anim7.pause();
				});
				Event.on('#button8', 'click', function(event) {
					alert(anim7.isPaused());
				});
				Event.on('#button9', 'click', function(event) {
					alert(anim7.isRunning());
				});
				Event.on('#button10', 'click', function(event) {
					anim7.resume();
				});
				Event.on('#button11', 'click', function(event) {
					anim7.stop();
				});
				Event.on('#button12', 'click', function(event) {
					anim7.stop(true);
				});
			});
		});
